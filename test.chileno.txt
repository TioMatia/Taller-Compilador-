// Definir función para calcular el factorial de un número
pega factorial(n) {
    si_po (n == 0) {
        devuelve_la_wa 1;
    } si_no {
        devuelve_la_wa n * factorial(n - 1);
    }
}

// Programa principal
int numero = 0;
int resultado;
string mensaje;

imprime_la_wa("Ingrese un número para calcular su factorial:");
lee numero;

resultado = factorial(numero);

si_po (resultado > 1000) {
    mensaje = "El factorial es muy grande";
} si_no {
    mensaje = "El factorial es: ";
}

imprime_la_wa(mensaje);
imprime_la_wa(resultado);

pa_cada (int i = 0; i < 5; i = i + 1) {
    imprime_la_wa("Contador: " + i);
}

mientras_la_wa (numero > 0) {
    imprime_la_wa(numero);
    numero = numero - 1;
}
