pega duplicar(x) {
    devuelve_la_wa x * 2;
}

// Definición de función que imprime si un número es par o impar
pega es_par(n) {
    si_po (n * 2 == 0) {
        imprime_la_wa("Es par");
    } si_no {
        imprime_la_wa("Es impar");
    }
}
